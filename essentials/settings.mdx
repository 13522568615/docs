每个Mintlify网站都需要一个“mint”。包含核心配置设置的Json文件。了解更多有关 [properties](#properties) 下面.

## 属性

<ResponseField name="name" type="string" required>
  项目的名称。用于全局标题。

  示例: `mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  包含该组内所有页面的组数组。

  <Expandable title="导航">
    <ResponseField name="group" type="string">
      组的名称。

      示例: `设置`
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      作为页面的Markdown文件的相对路径。

      示例: `["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Logo图像的路径或包含“light”和“dark”模式Logo图像路径的对象。

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      亮模式下的Logo路径。
    </ResponseField>

    <ResponseField name="dark" type="string">
      暗模式下的Logo路径。
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      点击Logo时链接到的位置。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Favicon图像的路径。
</ResponseField>

<ResponseField name="colors" type="Colors">
  您全局主题的十六进制颜色代码。

  <Expandable title="颜色">
    <ResponseField name="primary" type="string" required>
      主颜色。通常用于高亮内容、章节标题、强调，在亮模式下使用。
    </ResponseField>

    <ResponseField name="light" type="string">
      暗模式下的主颜色。通常用于高亮内容、章节标题、强调，在暗模式下使用。
    </ResponseField>

    <ResponseField name="dark" type="string">
      重要按钮的主颜色。
    </ResponseField>

    <ResponseField name="background" type="object">
      亮模式和暗模式下的背景颜色。

      <Expandable title="对象">
        <ResponseField name="light" type="string" required>
          亮模式下背景的十六进制颜色代码。
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          暗模式下背景的十六进制颜色代码。
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  您希望在顶部栏中包含的链接的`name`和`url`数组。

  <Expandable title="顶部链接">
    <ResponseField name="name" type="string">
      按钮的名称。

      示例: `联系我们`
    </ResponseField>

    <ResponseField name="url" type="string">
      点击按钮后链接到的URL。示例: `https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="顶部行动呼吁">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      链接显示一个按钮。GitHub显示提供的URL中的仓库信息，包括GitHub星标数量。
    </ResponseField>

    <ResponseField name="url" type="string">
      如果是`link`：按钮链接到的位置。

      如果是`github`：链接到加载GitHub信息的仓库。
    </ResponseField>

    <ResponseField name="name" type="string">
      按钮内的文本。仅在`type`为`link`时必需。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅在您希望在导航栏中显示不同版本的文档时使用。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  包含`icon`、`color`和`url`的锚点数组。

  <Expandable title="锚点">
    <ResponseField name="icon" type="string">
      用于突出显示锚点的[Font Awesome](https://fontawesome.com/search?q=heart)图标。

      示例: `comments`
    </ResponseField>

    <ResponseField name="name" type="string">
      锚点标签的名称。

      示例: `社区`
    </ResponseField>

    <ResponseField name="url" type="string">
      标记锚点中页面的URL开头。通常，这是您放置页面的文件夹名称。
    </ResponseField>

    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。也可以是渐变，如果您传递一个包含`from`和`to`属性的对象，每个属性都是一个十六进制颜色。
    </ResponseField>

    <ResponseField name="version" type="string">
      如果您希望在选择正确的文档版本之前隐藏锚点，则使用此字段。
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      如果您希望在直接链接到其中的文档之前隐藏锚点，请传递`true`。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      选项之一：“brands”、“duotone”、“light”、“sharp-solid”、“solid”或“thin”。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最上方锚点的默认配置。

  <Expandable title="对象">
    <ResponseField name="name" default="Documentation" type="string">
      最上方锚点的名称。
    </ResponseField>

    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome图标。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      选项之一：“brands”、“duotone”、“light”、“sharp-solid”、“solid”或“thin”。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航标签的数组。

  <Expandable title="标签">
    <ResponseField name="name" type="string">
      标签名称。
    </ResponseField>

    <ResponseField name="url" type="string">
      标记标签中页面的URL开头。通常，这是您放置页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API设置的配置。有关API页面的更多信息，请参见[API组件](/api-playground/demo)。

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有API端点的基本URL。如果`baseUrl`是一个数组，将启用多个基本URL选项，用户可以切换。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="认证">
        <ResponseField name="method" type="&#x22;bearer&#x22; | &#x22;basic&#x22; | &#x22;key&#x22;">
          用于所有API端点的认证策略。
        </ResponseField>

        <ResponseField name="name" type="string">
          在API游乐场中使用的认证参数名称。

          如果方法为`basic`，格式应为`[usernameName]:[passwordName]`。
        </ResponseField>

        <ResponseField name="inputPrefix" type="string">
          设计为认证输入字段前缀的默认值。

          例如，如果`inputPrefix`为`AuthKey`，则认证字段的默认输入结果将继承为`AuthKey`。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API游乐场的配置。

      <Expandable title="游乐场">
        <ResponseField name="mode" default="show" type="&#x22;show&#x22; | &#x22;simple&#x22; | &#x22;hide&#x22;">
          游乐场是否显示、隐藏，或仅显示没有用户交互的端点`simple`。

          在[游乐场指南](/api-playground/demo)中了解更多信息。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用此标志可确保OpenAPI页面中的键顺序与OpenAPI文件中定义的键顺序匹配。

      <Warning>此行为将很快默认启用，届时此字段将被弃用。</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  指向您的OpenAPI文件的URL或相对路径的字符串或字符串数组。

  示例：

  <CodeGroup>
    ```json 绝对
    "openapi": "https://example.com/openapi.json"
    ```

    ```json 相对
    "openapi": "/openapi.json"
    ```

    ```json 多个
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  社交媒体账户的对象，其中键:属性对表示社交媒体平台和账户URL。

  示例：

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      以下值之一`website`、`facebook`、`x`、`discord`、`slack`、`github`、`linkedin`、`instagram`、`hacker-news`。

      示例: `x`
    </ResponseField>

    <ResponseField name="property" type="string">
      社交平台的URL。

      示例: `https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  启用反馈按钮的配置。

  <Expandable title="反馈">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      启用一个按钮，允许用户通过拉取请求建议编辑。
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default="false">
      启用一个按钮，允许用户提出有关文档的问题。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  自定义暗模式切换。

  <Expandable title="模式切换">
    <ResponseField name="default" type={'"light" or "dark"'}>
      设置是否始终为新用户显示亮或暗模式。如果未设置，我们默认与用户的操作系统相同的模式。
    </ResponseField>

    <ResponseField name="isHidden" type="boolean" default="false">
      设置为true以隐藏暗/亮模式切换。您可以将`isHidden`与`default`结合使用，以强制您的文档仅使用亮或暗模式。例如：

      <CodeGroup>
        ```json 仅暗模式
        "modeToggle": {
          "default": "dark",
          "isHidden": true
        }
        ```

        ```json 仅亮模式
        "modeToggle": {
          "default": "light",
          "isHidden": true
        }
        ```
      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  每个页面后面显示的背景图像。请参见[Infisical](https://infisical.com/docs)和[FRPC](https://frpc.io)的示例。
</ResponseField>
